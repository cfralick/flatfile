FlatFile 0.0.0

Read and write tsv & csv.

Install 

$ composer install cfralick/flatfile

Basic Usage 

<?php
require_once __DIR__ . '/../vendor/autoload.php';
use FlatFile\CsvFile;

$csv = new CsvFile('file.csv');

$csv->writeRow(['1','2','3']);
// writes "1","2","3"
